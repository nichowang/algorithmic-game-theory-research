\section*{Algorithm}

\begin{equation*}
\text{Allocation}_i = \min\left(\frac{c_i}{2}, \lambda\right), \quad \text{where } \sum_{i=1}^{n} \min\left(\frac{c_i}{2}, \lambda\right) = E
\end{equation*}
