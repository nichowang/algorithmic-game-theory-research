\section{算法说明}

CG-CEL (Contested Garment CEL) 算法分两个阶段:

\subsection{规则}
设总财产为 $E$，索赔向量为 $c = (c_1, c_2, \ldots, c_n)$，令 $S = \sum_{i=1}^{n} c_i$ (总索赔)。

\subsubsection{情况1: $E \leq \frac{S}{2}$}
使用 \textbf{CEA} (Constrained Equal Awards) 分配在 $\frac{c_i}{2}$ 上:
\begin{itemize}
    \item 找到 $\lambda$ 使得 $\sum_{i=1}^{n} \min\left(\frac{c_i}{2}, \lambda\right) = E$
    \item 分配: $x_i = \min\left(\frac{c_i}{2}, \lambda\right)$
\end{itemize}

\subsubsection{情况2: $E > \frac{S}{2}$}
\begin{itemize}
    \item 第一步: 每人先获得 $\frac{c_i}{2}$
    \item 第二步: 剩余 $E - \frac{S}{2}$ 用 \textbf{CEL} (Constrained Equal Losses) 分配在 $\frac{c_i}{2}$ 上
    \item CEL: 找到 $\mu$ 使得 $\sum_{i=1}^{n} \max\left(0, \frac{c_i}{2} - \mu\right) = E - \frac{S}{2}$
    \item 最终分配: $x_i = \frac{c_i}{2} + \max\left(0, \frac{c_i}{2} - \mu\right)$
\end{itemize}

\section{$E=55$ 的情况验证}

\subsection{案例1: Claims=$[25, 25, 25, 25]$, $E=55$}

\begin{align*}
S &= 25 + 25 + 25 + 25 = 100 \\
\frac{S}{2} &= 50 \\
E &= 55 > 50 \quad \text{(使用情况2)}
\end{align*}

第一步: 每人先得 $\frac{25}{2} = 12.5$，总共先分配 $4 \times 12.5 = 50$

剩余: $55 - 50 = 5$

CEL阶段: 在 $(12.5, 12.5, 12.5, 12.5)$ 上分配5，找 $\mu$:
\begin{align*}
\sum \max(0, 12.5 - \mu) &= 5 \\
4(12.5 - \mu) &= 5 \\
50 - 4\mu &= 5 \\
\mu &= 11.25
\end{align*}

最终分配:
\begin{align*}
x_i = 12.5 + (12.5 - 11.25) = 12.5 + 1.25 = \boxed{13.75}
\end{align*}

\textbf{结果:} $[13.75, 13.75, 13.75, 13.75]$ \textcolor{green}{✓}

\subsection{案例2: Claims=$[30, 30, 20, 20]$, $E=55$}

\begin{align*}

\begin{align*}

\end{align*}

\begin{align*}

\end{align*}

最终:
\begin{align*}
x_1 &= 15 + (15 - 12.5) = \boxed{17.5} \\
x_2 &= 15 + (15 - 12.5) = \boxed{17.5} \\
x_3 &= 10 + \max(0, 10 - 12.5) = \boxed{10} \\
x_4 &= 10 + 0 = \boxed{10}
\end{align*}

\textbf{结果:} $[17.5, 17.5, 10, 10]$ \textcolor{green}{✓}
